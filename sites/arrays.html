For Each

const paragraph = document.querySelector('#sample-1')
var cars = ["Audi", "BMW", "Mercedes", "Volvo"]
cars.forEach(displayCars)

function displayCars(item, index) {
    paragraph.innerHTML += index + ": " + item + "<br>"
}


const links = document.querySelectorAll('.ForEachSample a')

function changeToBlack() {                          
    links.forEach((links) => {
        links.style.color = "black";                // change text color for every link
    })
}

links.forEach((links, index) => {
    links.addEventListener('click', function() {
        console.log("this: " + this)                // displays curent link in console
        console.log("index: " + index)              // display index of curent link in console
        changeToBlack()                             // call changeToBlack function
        this.style.color = "blue"                   // change text color of curent link
    })
})
